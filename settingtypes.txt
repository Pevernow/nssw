#    ABOUT SETTINGS

# Mapgen: write here (after "int") 6 (mgv6) or 7 (mgv7 or valleys or fractal or flat)
nssw.mymapgenis (Mapgen) int 7

# Mobs-Multiplier: multiplier defining the spawn abundance for the mobs:
# 0 = spawn disabled
# a number between 0 and 1000 = less then default
# 1000 = default
# more than 1000 = suicide is not the fastest way of dying
nssw.multimobs (Multimobs) int 1000

# Difficult-level:
# Multiplier to raise life and damage of the mobs
nssw.mob_difficulty (Mob difficulty) int 1

nssw.server_rainbow_staff (Use the server-mode rainbow staff) bool false

nssw.energy_boosts (Eat energy for stats boost) bool false
nssw.energy_lights (Place energy for light) bool false

nssw.spearmodifier (Spear speed modifier, 1=normal, 0.5 is half speed) string 1.0
